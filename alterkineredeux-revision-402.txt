  <script>
document.addEventListener("DOMContentLoaded", () => {
  const accordions = document.querySelectorAll('details.details');

  accordions.forEach(details => {
    const summary = details.querySelector('summary');

    summary.addEventListener('click', event => {
      event.preventDefault(); // Prevent instant open

      const isOpening = !details.hasAttribute('open');

      // First, close all others
      accordions.forEach(d => {
        if (d !== details) {
          d.style.height = '3.5em';
          d.removeAttribute('open');
        }
      });

      if (isOpening) {
        // Open this one
        details.setAttribute('open', 'true');
        const fullHeight = details.scrollHeight;
        details.style.height = '3.5em'; // Start small
        requestAnimationFrame(() => {
          details.style.height = fullHeight + 'px';
        });
      } else {
        // Close this one
        const currentHeight = details.offsetHeight;
        details.style.height = currentHeight + 'px';
        
        requestAnimationFrame(() => {
          details.style.height = '3.5em';
        });

        setTimeout(() => {
          details.removeAttribute('open');
        }, 400);
      }
    });
  });
});
</script>

<script>
function prepareForPrint() {
  // Expand all accordions before printing
  const accordions = document.querySelectorAll('details.details');
  
  accordions.forEach(details => {
    // Open all accordion items
    details.setAttribute('open', 'true');
    
    // Reset height styles to show all content
    details.style.height = 'auto';
    
    // Remove any rotating icons
    const summary = details.querySelector('.details__summary');
    if (summary) {
      summary.style.setProperty('--print-state', 'open');
    }
  });
  
  // Brief delay to allow DOM updates before printing
  setTimeout(() => {
    window.print();
  }, 500);
}
